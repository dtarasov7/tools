apiVersion: networking.k8s.io/v1

  namespace: fipa

  ingressClassName: nginx
  rules:
  - host: fipa.example.com
    http:
      paths:
      - backend:
          service:
            name: fipa-cip
            port:
              number: 443
        path: /
      - backend:
          service:
            name: fipa-cip
            port:
              number: 389
        path: /
        pathType: ImplementationSpecific
  tls:
  - hosts:
    - fipa.example.com
    secretName: fipa-test
